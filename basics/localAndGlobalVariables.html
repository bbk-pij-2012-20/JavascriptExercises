Basics exercise#2: code to demonstrate the behaviour of local and global variables
-------------------------------------------------------------------------------------------------

METHOD#1:

var method1 = function() {

  var n = 10;
  console.log(n);

}

method1() // gives 10
console.log(n) // gives ReferenceError because n was declared with var inside the method so it is local to the method.

-------------------------------------------------------------------------------------------------

METHOD#2:

var method2 = function() {

  var n = 10;
  n2 = 100;
  console.log(n);

  console.log(n2);

}

method2() // gives 10 and 100
console.log(n2) // gives 100 because n2 was not declared with var, so n2 is a global variable.

console.log(n) // gives ReferenceError because n was declared with var inside the method so it is local to the method.

-------------------------------------------------------------------------------------------------
