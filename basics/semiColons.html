Exercise#1: code to demonstrate what can happen in the absence of semi-colons
-------------------------------------------------------------------------------------------------

METHOD#1:

var method1 = function(n) {

  if (n < 10) {

    return n * 2;

  } else {

    return n;

  }

}

console.log(method1(14)) gives 14
console.log(method1(3)) gives 6

-------------------------------------------------------------------------------------------------

METHOD#2:

var method2 = function(n) {

  if (n < 10) {

    return n * 2 // with no semi-colon, but still all on one line. Javascript engine puts the semi-colon at the end of this statement, hence it returns 6.

  } else {

    return n;

  }

}

console.log(method2(14)) gives 14
console.log(method2(3)) gives 6

-------------------------------------------------------------------------------------------------

METHOD#3:

var method3 = function(n) {

  if (n < 10) {

    return // here there is no semi-colon. The rest of the statement is on the next line.
     n * 2 // Because this expression suffices as a complete statement in its own right,
             //Javascript engine puts a semi-colon at the end of the line above, hence it returns undefined.

  } else {

    return n;

  }

}

console.log(method3(14)) gives 14
console.log(method3(3)) gives undefined

-------------------------------------------------------------------------------------------------

METHOD#4:

var method4 = function(n) {

  if (n < 10) {

    return n * // again there is no semi-colon. The rest of the statement is on the next line.
     2           // This time because the expression on line above is incomplete, Javascript engine does not put a semi-colon at the end of the line above, hence it returns 6.

  } else {

    return n;

  }

}

console.log(method4(14)) gives 14
console.log(method4(3)) gives undefined
