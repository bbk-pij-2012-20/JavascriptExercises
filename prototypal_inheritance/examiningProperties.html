examining properties with for loop

var Bike = function(year) {

       console.log("..called Bike constructor..");

       this.year = year;

}

Bike.prototype.km = 0;

Bike.prototype.pedal = function(km_travelled) {

       console.log("..you're cycling.. ");
       this.km = this.km + km_travelled;

}

var bike1 = new Bike(2016);

for (var prop in bike1) {

       console.log(prop + " " + bike1[prop]);  // loops through the properties of bike1, but this also includes the properties in its prototype (e.g. pedal())

}

var propName = 'year';  // this is another way to get a property from the object.
console.log(bike1[propName]);  // square brackets as though a hashmap (i.e. name-value pair)
