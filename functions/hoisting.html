Functions exercise#1: hoisting
-------------------------------------------------------------------------------------------------

Example#1: hoisting

function nutella(n){

          console.log("Nutella has nuts numbering " + n);

};

nutella(10); //not as expected, gives the console.log() output of the 2nd function definition (below)

function nutella(n){

          console.log("...but Nutella now has nuts numbering " + n);

};

nutella(20); //as expected, gives the console.log() output of the 2nd function definition

-------------------------------------------------------------------------------------------------

Example#2: no hoisting with preferred syntax

var nutella = function(n){

          console.log("Nutella has nuts numbering " + n);

};

nutella(10); //as expected, gives the console.log() output of the 1st function definition (above)

nutella = function(n){

          console.log("...but Nutella now has nuts numbering " + n);

};

nutella(10); //as expected, gives the console.log() output of the 2nd function definition

-------------------------------------------------------------------------------------------------
